87 @staticmethod 

88 def copy_artifacts(workflow_name, artifact_files): 

89 """ 

90 copy workflow artifacts in a common workflow directory for archiving 

91 :param workflow_name: name of workflow being archived 

92 :param artifact_files: list of files to be copied in archive 

93 :return: 

94 """ 

95 workflow_path = os.path.join(tempfile.gettempdir(), workflow_name) 

96 if os.path.exists(workflow_path): 

97 shutil.rmtree(workflow_path) 

98 WorkflowExportUtils.make_dir(workflow_path) 

99 for path in artifact_files: 

100 if path: 

101 for file in path.split(","): 

102 shutil.copy(file, workflow_path) 

103 

104 return workflow_path

106 @staticmethod 

107 def archive(export_file, workflow_name, workflow_path, manifest): 

108 """ 

109 Create a workflow-archive 

110 :param export_file: 

111 :param workflow_name: 

112 :param workflow_path 

113 :param manifest: 

114 :return: 

115 """ 

116 war_path = WorkflowExportUtils.get_archive_export_path(export_file, workflow_name) 

117 try: 

118 with zipfile.ZipFile(war_path, 'w', zipfile.ZIP_DEFLATED) as z: 

119 WorkflowExportUtils.archive_dir(workflow_path, z) 

120 # Write the manifest here now as a json 

121 z.writestr(os.path.join(WAR_INF, MANIFEST_FILE_NAME), manifest) 

122 except IOError: 

123 logging.error("Failed to save the workflow-archive to workflow-path \"%s\". " 

124 "Check the file permissions and retry.", export_file) 

125 raise 

126 except Exception as e: 

127 logging.error("Failed to convert %s to the workflow-archive.", workflow_name) 

128 raise e

130 @staticmethod 

131 def archive_dir(path, dst): 

132 

133 """ 

134 This method zips the dir and filters out some files based on a expression 

135 :param path: 

136 :param dst: 

137 :return: 

138 """ 

139 unwanted_dirs = {'__MACOSX', '__pycache__'} 

140 

141 for root, directories, files in os.walk(path): 

142 # Filter directories 

143 directories[:] = [d for d in directories if WorkflowExportUtils.directory_filter(d, unwanted_dirs)] 

144 for f in files: 

145 file_path = os.path.join(root, f) 

146 dst.write(file_path, os.path.relpath(file_path, path))
